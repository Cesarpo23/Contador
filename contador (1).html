<!DOCTYPE html>
<html>
<head>
<style>
#contador {
font-size: 50px;
color: red;
text-align: center;
}
</style>
</head>
<body>
<div id="contador">0</div>
<script>
// Crear una variable con la fecha y hora objetivo
var fechaObjetivo = new Date('2024-02-21 00:00:00');
// Obtener el elemento HTML donde se muestra el contador
var elemento = document.getElementById("contador");
// Crear una función que se ejecuta cada segundo
setInterval(function() {
// Obtener la fecha y hora actuales
var fechaActual = new Date();
// Calcular la diferencia en milisegundos
var diferencia = fechaObjetivo - fechaActual;
// Si la diferencia es positiva, calcular los días, horas, minutos y segundos
if (diferencia > 0) {
var dias = Math.floor(diferencia / (1000 * 60 * 60 * 24));
var horas = Math.floor((diferencia % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
var minutos = Math.floor((diferencia % (1000 * 60 * 60)) / (1000 * 60));
var segundos = Math.floor((diferencia % (1000 * 60)) / 1000);
// Añadir un cero si es necesario
if (dias < 10) {
dias = "0" + dias;
}
if (horas < 10) {
horas = "0" + horas;
}
if (minutos < 10) {
minutos = "0" + minutos;
}
if (segundos < 10) {
segundos = "0" + segundos;
}
// Actualizar el elemento HTML con el nuevo valor
elemento.innerHTML = dias + ":" + horas + ":" + minutos + ":" + segundos;
} else {
// Si la diferencia es negativa o cero, mostrar un mensaje
elemento.innerHTML = "¡Ha llegado el día!";
}
}, 1000); // 1000 milisegundos = 1 segundo
</script>
</body>
